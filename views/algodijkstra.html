const  int  infinity  =  1e9;

    void  dijkstra(void){
        
        int  n,  m;
        
        cin  >>  n  >>  m;
        
        map  <pair  <int,  int>  ,  int>  multiple_edges;
        
        vector  <int>  adj[n  +  1];
        
        while(m--){
            
            int  u,  v,  w;
            
            cin  >>  u  >>  v  >>  w;
            
            adj[u].pb(v);
            
            adj[v].pb(u);
            
            int  A  =  min(u,  v);
            
            int  B  =  max(u,  v);
            
            if(multiple_edges.find(mp(A,  B))  ==  multiple_edges.end()){
                
                multiple_edges[mp(A,  B)]  =  w;
                
            }
            else{
                
                multiple_edges[mp(A,  B)]  =  min(multiple_edges[mp(A,  B)],  w);
                
            }
            
        }
        
        map  <int,  int>  parent;
        
        int  source  =  1;
        
        vector  <int>  dist(n  +  1,  infinity);
        
        dist[source]  =  0;
        
        priority_queue<pair<int,  int>,  vector<pair<int,  int>  >,  greater<pair<int,  int>  >  >  pq;
        
        pq.push({dist[source],  source});
        
        while(!pq.empty()){
            
            int  u  =  pq.top().second;
            
            pq.pop();
            
            for(int  v  :  adj[u]){
                
                int  w  =  multiple_edges[mp(min(u,  v),  max(u,  v))];
                
                if(dist[u]  +  w  <  dist[v]){
                    
                    parent[v]  =  u;
                    
                    dist[v]  =  dist[u]  +  w;
                    
                    pq.push({dist[v],  v});
                    
                }
                
            }
            
        }

        if(dist[n]  ==  infinity){
            
            cout  <<  -1  <<  endl;
            
            return;
            
        }

        vector  <int>  path;

        int  now  =  n;

        path.pb(n);

        while(parent[now]  !=  source){
            
            path.pb(parent[now]);
            
            now  =  parent[now];
            
        }

        path.pb(1);

        reverse(path.begin(),  path.end());
        
        for(int  x  :  path)  {
            
            cout  <<  x  <<  "  ";
            
        }

    }
